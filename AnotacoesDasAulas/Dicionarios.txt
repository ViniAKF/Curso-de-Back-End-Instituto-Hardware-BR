"""
Dicionarios aninhados 
- Permite que olhe as variaveis individualizando cada conjunto de dados por uma outra chave
"""
nome="nome" #Teste
contatos = {"Junior@gmail.com": {f"{nome}":"Junior","idade": 25}, "Amanda@gmail.com" : {"nome":"Amanda", "idade":25}}

#Para iterar os dicionários faz com o 'for' mesmo
for chave in contatos:
    print(chave,contatos[chave])

for chave,valor in contatos.items(): #Faz isso para que possa pegar tanto a chave quanto o valor direto
